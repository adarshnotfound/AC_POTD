class Solution {
    public int findGCD(int[] nums) {
        int min=nums[0],max=nums[0],gcd=0;
        for(int i=1;i<nums.length;i++){
            if(nums[i]<min)
              min=nums[i];
            if(nums[i]>max)
              max=nums[i];
        }
        for(int i=1;i<=Math.min(min,max);i++){
            if(min%i==0&&max%i==0)
            gcd=i;
        }
        return gcd;
    }
}
