class Solution {
    public String addStrings(String num1, String num2) {
        int i1=num1.length()-1;
        int i2=num2.length()-1;
        String sum="";
        int carry=0;
        while(i1>=0||i2>=0){
            int a=0;
            if(i1!=-1) a+=(num1.charAt(i1--)-48);
            if(i2!=-1) a+=(num2.charAt(i2--)-48);
            sum=(a+carry)%10+sum;
            carry=(a+carry)/10;
        }
        if(carry!=0) return carry+sum;
        return sum;
    }
}
