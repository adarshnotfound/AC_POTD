class Solution {
    public void rotate(int[] nums, int k) {
        int len=nums.length;
        reverse(nums,0,len-1);
        k=k<len?k:k%len;
        reverse(nums,0,k-1);
        reverse(nums,k,len-1);
    }
    public void reverse(int a[],int start, int end){
        int temp;
        while(start<end){
            temp=a[end];
            a[end]=a[start];
            a[start]=temp;
            start++;
            end--;
        }
    }
}
