import java.util.*;
public class Main {
    public static void main(String[] args) {
      Scanner sc=new Scanner(System.in);
      int k=sc.nextInt();
      int w=sc.nextInt();
      int n=sc.nextInt();
      int total=0;
      for(int i=1;i<=n;i++)
        total+=(i*k);
      int toBorrow=total-w;
      if(toBorrow<=0)
        toBorrow=0;
      System.out.println(toBorrow);
  }
}
